     function generateUUID() {
  var d = new Date().getTime();
  var uuid = 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(
    c

    var r = ((d + Math.random() * 16) % 16) | 0;
    d = Math.floor(d / 16);

  });
  return uuid;
}
